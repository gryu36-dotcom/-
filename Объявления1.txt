<script>
const ANNOUNCEMENTS = [
  {name:"Запрещается!", cat:"Запреты", url:"https://raw.githubusercontent.com/gryu36-dotcom/-/main/zapreshaetsy.mp3"},
  {name:"Тарелки", cat:"Запреты", url:"https://raw.githubusercontent.com/gryu36-dotcom/-/main/Tarelki.mp3"},
  {name:"Тапочки", cat:"Запреты", url:"https://raw.githubusercontent.com/gryu36-dotcom/-/main/Tapki.mp3"},
  {name:"Зонты", cat:"Экстренные", url:"https://raw.githubusercontent.com/gryu36-dotcom/-/main/zonti.mp3"},
  {name:"За час до закрытия", cat:"Закрытие", url:"https://raw.githubusercontent.com/gryu36-dotcom/-/main/za%20CHas.mp3"},
  {name:"За полчаса до закрытия", cat:"Закрытие", url:"https://raw.githubusercontent.com/gryu36-dotcom/-/main/za30min.mp3"},
  {name:"Кукуруза", cat:"Реклама", url:null},
  {name:"Напитки", cat:"Реклама", url:null},
  {name:"Плавательные аксессуары", cat:"Реклама", url:null},
  {name:"Горки", cat:"Приглашения", url:"https://raw.githubusercontent.com/gryu36-dotcom/-/main/Gorki.mp3"},
];

const categories = [...new Set(ANNOUNCEMENTS.map(a => a.cat))];
let activeFilter = "Все";
let currentIndex = null;

const filtersEl = document.getElementById('filters');
const listEl = document.getElementById('list');
const volumeSlider = document.getElementById('volume');
const volValue = document.getElementById('volValue');

const audio = new Audio();
audio.volume = volumeSlider.value / 100;

function fmtTime(s){
  if(!isFinite(s) || s < 0) s = 0;
  const m = Math.floor(s/60);
  const sec = Math.floor(s%60).toString().padStart(2,'0');
  return `${m}:${sec}`;
}

function buildFilters(){
  const counts = {};
  ANNOUNCEMENTS.forEach(a => counts[a.cat] = (counts[a.cat]||0)+1);
  filtersEl.appendChild(makeFilterBtn("Все", ANNOUNCEMENTS.length));
  categories.forEach(cat => filtersEl.appendChild(makeFilterBtn(cat, counts[cat])));
}

function makeFilterBtn(label, count){
  const btn = document.createElement('button');
  btn.className = 'filter-btn' + (label === activeFilter ? ' active' : '');
  btn.innerHTML = `${label}<span class="count">${count}</span>`;
  btn.addEventListener('click', () => {
    activeFilter = label;
    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    renderList();
  });
  return btn;
}

function renderList(){
  listEl.innerHTML = '';
  const filtered = ANNOUNCEMENTS
    .map((a,i) => ({...a, idx:i}))
    .filter(a => activeFilter === "Все" || a.cat === activeFilter);

  if(filtered.length === 0){
    listEl.innerHTML = '<div class="empty-state">В этом разделе пока нет роликов.</div>';
    return;
  }

  let lastCat = null;
  filtered.forEach(item => {
    if(activeFilter === "Все" && item.cat !== lastCat){
      lastCat = item.cat;
      const label = document.createElement('div');
      label.className = 'section-label';
      label.textContent = item.cat;
      listEl.appendChild(label);
    }
    listEl.appendChild(makeTrackRow(item));
  });
}

function makeTrackRow(item){
  const row = document.createElement('div');
  const isActive = currentIndex === item.idx;
  row.className = 'track' + (item.url ? '' : ' missing') + (isActive ? ' playing' : '');
  row.dataset.idx = item.idx;

  const isPlayingNow = isActive && !audio.paused;
  const iconClass = isPlayingNow ? 'icon-pause' : 'icon-play';

  row.innerHTML = `
    <div class="track-row">
      <button class="play-btn" ${item.url ? '' : 'disabled'} aria-label="Воспроизвести" tabindex="-1">
        <span class="icon ${iconClass}"></span>
      </button>
      <div class="track-info">
        <div class="track-name">${item.name}</div>
        <div class="track-meta ${item.url ? '' : 'no-link'}">${item.cat}${item.url ? '' : ' · ссылка не указана'}</div>
      </div>
      <div class="track-bars"><span class="bar"></span><span class="bar"></span><span class="bar"></span></div>
    </div>
    <div class="track-player">
      ${isActive ? `
      <div class="seek-row">
        <span class="time cur">0:00</span>
        <input type="range" class="seek" min="0" max="100" value="0" step="0.1">
        <span class="time end">0:00</span>
      </div>` : ''}
    </div>
  `;

  if(item.url){
    row.querySelector('.track-row').addEventListener('click', () => toggleTrack(item.idx));
  }

  if(isActive){
    const seek = row.querySelector('.seek');
    seek.addEventListener('input', () => {
      const pct = seek.value;
      seek.style.setProperty('--fill', pct + '%');
      if(audio.duration){
        audio.currentTime = (pct/100) * audio.duration;
      }
    });
    row._seek = seek;
    row._curTime = row.querySelector('.time.cur');
    row._endTime = row.querySelector('.time.end');
    updateProgressUI(row);
  }

  return row;
}

function updateProgressUI(row){
  if(!row || !row._seek) return;
  const dur = audio.duration || 0;
  const cur = audio.currentTime || 0;
  const pct = dur ? (cur/dur*100) : 0;
  row._seek.value = pct;
  row._seek.style.setProperty('--fill', pct + '%');
  row._curTime.textContent = fmtTime(cur);
  row._endTime.textContent = fmtTime(dur);
}

function getActiveRow(){
  return listEl.querySelector(`.track[data-idx="${currentIndex}"]`);
}

function toggleTrack(idx){
  const item = ANNOUNCEMENTS[idx];
  if(!item.url) return;

  if(currentIndex === idx){
    if(audio.paused){ audio.play(); } else { audio.pause(); }
    renderList();
    return;
  }

  currentIndex = idx;
  audio.src = item.url;
  audio.play().catch(() => {});
  renderList();
}

audio.addEventListener('timeupdate', () => updateProgressUI(getActiveRow()));
audio.addEventListener('loadedmetadata', () => updateProgressUI(getActiveRow()));
audio.addEventListener('play', renderList);
audio.addEventListener('pause', renderList);
audio.addEventListener('ended', () => { renderList(); });

volumeSlider.addEventListener('input', () => {
  const v = volumeSlider.value;
  audio.volume = v / 100;
  volValue.textContent = v + '%';
});

buildFilters();
renderList();
</script>
